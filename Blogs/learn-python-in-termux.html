Your first real script (useful + motivating)

Make a script that creates a folder in Downloads and saves notes. This teaches files, strings, and arguments.

mkdir -p ~/Projects/python-notes
cd ~/Projects/python-notes
cat > notes.py << 'PY'
import os, sys, datetime

base = os.path.expanduser("~/storage/shared/Download/Python Notes")
os.makedirs(base, exist_ok=True)

text = " ".join(sys.argv[1:]).strip() or "Empty note"
stamp = datetime.datetime.now().strftime("%Y-%m-%d_%H-%M-%S")
path = os.path.join(base, f"note_{stamp}.txt")

with open(path, "w", encoding="utf-8") as f:
    f.write(text + "\n")

print("Saved:", path)
PY
python notes.py "Hello from Termux!"
